    x_max = max(x1data)

    active_dots = 20

    if (x_range is None):

    tmax = len(y1data) - 1 - step_size

        while t < tmax:

    def run(data):
        # update the data
        x, V = data
        if (type(x) == type(x1data[-1]) == list and all(
                [xi == x1i for xi, x1i in zip(x, x1data[-1])])):  # last step of traj
